var nameField = screen.find('Nombre');
var telefonoField = screen.find('Telefono');
// var emailField = screen.find('Email');
var http = new XMLHttpRequest();
var formData = new FormData();

// "Current web app" URL we got after the deploy
var googleAppURL = 'http://backend.devcodegt.com/influmedia/public/api/nissan';

// formData keys need to match column names in the spreadsheet (case sensitive)
// formData.append('Email', emailField.value);
formData.append('telefono', telefonoField.value);
formData.append('nombre', nameField.value);

// define what happens on successful form data submission
http.addEventListener('load', function (evt) {
    // call 'c' because the action is considered "completed".
    c();
});
// define what happens in case of an error
http.addEventListener('error', function (evt) {
    console.log('Error!');
});
// set up our request
http.open('POST', googleAppURL, true);
// send our data
http.send(formData);
